基础语法

Python 使用缩进来定义代码块，告别大括号。

Python
# 变量与数据类型
name = "Python"           # 字符串
age = 30                   # 整数
pi = 3.14159              # 浮点数
is_fun = True            # 布尔值
nothing = None            # 空值

# 输入输出
name = input("请输入名字: ")
print(f"你好, {name}!")    # f-string 格式化

# 条件语句
if age >= 18:
    print("成年")
elif age >= 12:
    print("青少年")
else:
    print("儿童")

# 循环
for i in range(5):        # 0, 1, 2, 3, 4
    print(i)

while count < 5:
    print(count)
    count += 1

# 列表推导式
squares = [x ** 2 for x in range(10)]
# [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]
📘 注意Python 使用缩进（通常 4 个空格）而不是大括号来定义代码块。缩进错误会导致 IndentationError。

内置数据结构

Python 提供了丰富且强大的内置数据结构。

Python
# 列表 (list) - 有序可变序列
fruits = ["苹果", "香蕉", "橙子"]
fruits.append("葡萄")     # 添加元素
fruits.sort()               # 排序
first, *rest = fruits      # 解包

# 元组 (tuple) - 有序不可变
point = (3, 4)
x, y = point              # 元组解包

# 字典 (dict) - 键值对
user = {"name": "Alice", "age": 25}
user["email"] = "alice@example.com"
for key, value in user.items():
    print(key, value)

# 集合 (set) - 无序不重复
tags = {"python", "编程", "python"}  # {"python", "编程"}
tags & {"python", "java"}         # 交集: {"python"}
tags | {"go"}                     # 并集

函数与模块

Python
# 函数定义
def greet(name: str, times: int = 1) -> str:  # 类型注解
    """返回问候语"""                       # 文档字符串
    return f"你好, {name}! " * times

# 关键字参数
greet(times=3, name="Bob")

# 可变参数
def sum_all(*args):              # args 是元组
    return sum(args)

def print_info(**kwargs):         # kwargs 是字典
    for k, v in kwargs.items():
        print(f"{k}: {v}")

# Lambda 匿名函数
square = lambda x: x ** 2
numbers = [1, 2, 3, 4]
squared = list(map(lambda x: x ** 2, numbers))

# 模块导入
import math
from datetime import datetime
import numpy as np

面向对象编程

Python
class Animal:
    """动物基类"""
    def __init__(self, name):     # 构造方法
        self.name = name

    def speak(self):                 # 实例方法
        raise NotImplementedError

    @classmethod
    def create_unknown(cls):        # 类方法
        return cls("未知生物")

    @staticmethod
    def info():                     # 静态方法
        return "这是一个动物类"

    def __str__(self):               # 特殊方法
        return f"动物: {self.name}"


class Dog(Animal):                 # 继承
    def speak(self):                 # 方法重写
        return "汪汪!"

# 装饰器
def timer(func):
    import time
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"执行耗时: {time.time() - start:.4f}秒")
        return result
    return wrapper

@timer
def slow_function():
    import time
    time.sleep(1)

文件 I/O 与异常

Python
# 文件读写 (使用 with 自动关闭)
with open("file.txt", "r", encoding="utf-8") as f:
    content = f.read()           # 读取全部
    lines = f.readlines()        # 读取所有行
    for line in f:               # 逐行读取（大文件推荐）
        print(line.strip())

with open("output.txt", "w") as f:
    f.write("Hello, World!\n")
    f.writelines(["line1\n", "line2\n"])

# 异常处理
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"除数不能为0: {e}")
except Exception as e:
    print(f"未知错误: {e}")
else:
    print("没有异常发生")
finally:
    print("无论是否有异常都会执行")
