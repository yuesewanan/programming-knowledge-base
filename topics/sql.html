SQL

结构化查询语言，操作关系型数据库的标准语言。从简单的数据查询到复杂的数据分析，SQL 是数据世界的基础技能。

基础查询

SELECT 是 SQL 中最核心的语句，用于从表中检索数据。

SQL
-- 查询所有列
SELECT * FROM users;

-- 查询指定列
SELECT name, email, age FROM users;

-- 列别名
SELECT
    name AS "姓名",
    email AS "邮箱"
FROM users;

-- 去重
SELECT DISTINCT city FROM users;

-- 限制返回行数
SELECT * FROM products LIMIT 10;       -- MySQL, PostgreSQL
SELECT * FROM products FETCH FIRST 10 ROWS ONLY;  -- SQL Server, Oracle

-- 常量列
SELECT name, '活跃' AS status FROM users;

条件过滤与排序

SQL
-- WHERE 子句
SELECT * FROM users
WHERE age >= 18;

-- 多条件
SELECT * FROM orders
WHERE status = 'completed'
  AND total > 100
  AND (created_at >= '2024-01-01' OR priority = 'high');

-- IN 操作符
SELECT * FROM products
WHERE category IN ('电子产品', '书籍', '服装');

-- BETWEEN 范围
SELECT * FROM employees
WHERE salary BETWEEN 5000 AND 10000;

-- LIKE 模糊匹配
SELECT * FROM users
WHERE name LIKE '张%';      -- 以"张"开头

-- NULL 判断
SELECT * FROM users
WHERE phone IS NULL;         -- IS NOT NULL 判断非空

-- 排序
SELECT name, price
FROM products
ORDER BY price DESC, name ASC;    -- DESC 降序，ASC 升序（默认）

表连接 (JOIN)

JOIN 用于将多个表中的数据关联查询，是关系型数据库的核心操作。

表A (users) 表B (orders) ┌─────┬──────┐ ┌─────┬──────────┐ │ id │ name │ │ id │ user_id │ ├─────┼──────┤ ├─────┼──────────┤ │ 1 │ Alice│ │ 101 │ 1 │ │ 2 │ Bob │ │ 102 │ 1 │ │ 3 │ Carol│ │ 103 │ 3 │ └─────┴──────┘ └─────┴──────────┘ INNER JOIN → 只返回匹配的行 (用户1,3) LEFT JOIN → 返回左表所有行 (用户1,2,3)
SQL
-- INNER JOIN: 只返回两表匹配的行
SELECT u.name, o.total, o.created_at
FROM users u
INNER JOIN orders o ON u.id = o.user_id;

-- LEFT JOIN: 返回左表所有行，右表无匹配则为 NULL
SELECT u.name, COUNT(o.id) AS order_count
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.id, u.name;

-- RIGHT JOIN: 返回右表所有行
SELECT e.name, d.name AS department
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.id;

-- FULL JOIN: 返回所有行
-- (MySQL 不直接支持，可用 LEFT + UNION + RIGHT 模拟)

-- 自连接: 同一张表连接自己
SELECT e1.name AS employee,
       e2.name AS manager
FROM employees e1
LEFT JOIN employees e2 ON e1.manager_id = e2.id;

聚合与分组

SQL
-- 聚合函数
SELECT
    COUNT(*) AS total_users,           -- 计数
    AVG(age) AS avg_age,               -- 平均值
    SUM(amount) AS total_amount,       -- 总和
    MAX(score) AS max_score,            -- 最大值
    MIN(score) AS min_score             -- 最小值
FROM users;

-- GROUP BY 分组
SELECT
    department,
    COUNT(*) AS employee_count,
    ROUND(AVG(salary), 2) AS avg_salary
FROM employees
GROUP BY department;

-- HAVING: 对分组结果过滤 (WHERE 不行)
SELECT
    category,
    COUNT(*) AS product_count,
    AVG(price) AS avg_price
FROM products
GROUP BY category
HAVING COUNT(*) >= 5;      -- 只显示商品数 >= 5 的分类

-- WHERE vs HAVING
-- WHERE: 分组前过滤行
-- HAVING: 分组后过滤组

表操作 (DDL)

SQL
-- 创建表
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    age INT CHECK (age >= 0),
    city VARCHAR(100) DEFAULT '未知',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 修改表
ALTER TABLE users ADD COLUMN phone VARCHAR(20);
ALTER TABLE users DROP COLUMN phone;
ALTER TABLE users MODIFY COLUMN name VARCHAR(200);

-- 外键约束
CREATE TABLE orders (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    total DECIMAL(10, 2),
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE
);

-- 插入数据
INSERT INTO users (name, email, age)
VALUES ('Alice', 'alice@example.com', 25);

-- 更新数据
UPDATE users
SET age = 26, city = '北京'
WHERE id = 1;

-- 删除数据
DELETE FROM users WHERE id = 1;
TRUNCATE TABLE users;      -- 清空表 (不可回滚)

索引与事务

索引提升查询性能，事务确保数据一致性。

SQL
-- 创建索引
CREATE INDEX idx_users_email ON users(email);
CREATE UNIQUE INDEX idx_users_phone ON users(phone);
CREATE INDEX idx_orders_user_date ON orders(user_id, created_at);  -- 复合索引

-- 查看执行计划
EXPLAIN SELECT * FROM users WHERE email = 'test@test.com';

-- 事务 (ACID)
START TRANSACTION;

UPDATE accounts
SET balance = balance - 100
WHERE id = 1;

UPDATE accounts
SET balance = balance + 100
WHERE id = 2;

COMMIT;     -- 提交事务
-- ROLLBACK; -- 回滚事务

-- 事务隔离级别
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
⚡

索引类型

B+树索引（默认）、哈希索引、全文索引、空间索引。选择合适的索引类型对查询性能至关重要

🔒

ACID

原子性 (Atomicity)、一致性 (Consistency)、隔离性 (Isolation)、持久性 (Durability)

👁️

隔离级别

READ UNCOMMITTED → READ COMMITTED → REPEATABLE READ → SERIALIZABLE（从低到高）

📊

执行计划

使用 EXPLAIN（MySQL）/ EXPLAIN ANALYZE（PostgreSQL）分析查询性能瓶颈
