基础语法

JavaScript 支持多种变量声明方式和数据类型。

JavaScript 
// 变量声明
var oldWay = '函数作用域';     // 旧方式，避免使用
let name = 'Alice';              // 块级作用域，可重新赋值
const PI = 3.14159;               // 常量，不可重新赋值

// 数据类型
const str = 'Hello';              // 字符串
const num = 42;                    // 数字
const bool = true;                // 布尔值
const n = null;                    // 空值
const u = undefined;               // 未定义
const arr = [1, 2, 3];           // 数组
const obj = { key: 'value' };    // 对象

// 模板字符串
const greeting = `你好，${name}！今年 ${num} 岁？`;

// 条件与循环
if (num > 18) {
  console.log('成年');
} else {
  console.log('未成年');
}

for (let i = 0; i < 5; i++) {
  console.log(i);
}

// 三元运算符
const status = age >= 18 ? '成年' : '未成年';

函数

函数是 JavaScript 的一等公民，有多种定义方式。

JavaScript 
// 函数声明
function add(a, b) {
  return a + b;
}

// 函数表达式
const subtract = function(a, b) {
  return a - b;
};

// 箭头函数 (ES6)
const multiply = (a, b) => a * b;

// 默认参数
const greet = (name = '访客') => `你好，${name}！`;

// 剩余参数
const sum = (...numbers) => numbers.reduce((acc, n) => acc + n, 0);

// 闭包：函数可以访问其外部作用域的变量
function counter() {
  let count = 0;
  return function() {
    return ++count;
  };
}
const myCounter = counter();
console.log(myCounter());  // 1
console.log(myCounter());  // 2

对象与数组

对象和数组是 JavaScript 中最常用的数据结构。

JavaScript 
// 对象字面量
const user = {
  name: 'Alice',
  age: 25,
  skills: ['JS', 'Python'],
  sayHi() {
    return `Hi, I'm ${this.name}`;
  }
};

// 解构赋值
const { name, age } = user;
const [first, second] = ['a', 'b', 'c'];

// 展开运算符
const clone = { ...user, age: 26 };
const merged = [...[1, 2], ...[3, 4]];

// 可选链操作符 ?.
const city = user?.address?.city ?? '未知';

// 空值合并 ??
const score = input ?? 0;  // 只在 null/undefined 时使用默认值

DOM 操作

DOM（文档对象模型）允许 JavaScript 操作 HTML 元素的内容、属性和样式。

JavaScript 
// 选择元素
const el = document.getElementById('myId');
const els = document.querySelectorAll('.myClass');
const first = document.querySelector('div > p');

// 修改内容和属性
el.textContent = '新文本';       // 纯文本
el.innerHTML = '<strong>加粗</strong>';  // HTML
el.setAttribute('data-id', '123');
el.classList.add('active');
el.classList.remove('hidden');
el.classList.toggle('visible');

// 创建和删除元素
const div = document.createElement('div');
div.textContent = '新元素';
document.body.appendChild(div);

// 修改样式
el.style.color = 'red';
el.style.backgroundColor = '#f0f0f0';

事件处理

事件驱动是 JavaScript 的核心编程模型。

JavaScript 
// 添加事件监听
const btn = document.getElementById('myBtn');
btn.addEventListener('click', (event) => {
  console.log('按钮被点击！', event);
});

// 常用事件类型
// click, dblclick, mouseover, mouseout, mousedown, mouseup
// keydown, keyup, keypress
// submit, change, input, focus, blur
// scroll, resize, load, DOMContentLoaded

// 事件委托：利用事件冒泡，减少监听器数量
document.addEventListener('click', (e) => {
  if (e.target.matches('.delete-btn')) {
    console.log('删除按钮被点击', e.target);
  }
});

// 阻止默认行为
form.addEventListener('submit', (e) => {
  e.preventDefault();  // 阻止表单提交刷新页面
});

异步编程

JavaScript 是单线程语言，通过事件循环机制处理异步操作。

JavaScript 
// 回调函数
function fetchData(callback) {
  setTimeout(() => {
    callback('数据加载完成');
  }, 1000);
}

// Promise
const promise = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('成功');
    // reject(new Error('失败'));
  }, 1000);
});

promise
  .then(result => console.log(result))
  .catch(err => console.error(err))
  .finally(() => console.log('结束'));

// async / await
async function getUserData(userId) {
  try {
    const response = await fetch(`https://api.example.com/users/${userId}`);
    if (!response.ok) throw new Error('请求失败');
    const data = await response.json();
    return data;
  } catch (error) {
    console.error('获取用户数据失败:', error);
    throw error;
  }
}

// Promise 并行
const [users, posts] = await Promise.all([
  fetch('/api/users').then(r => r.json()),
  fetch('/api/posts').then(r => r.json())
]);
异步编程演进: ┌──────────┐ ┌──────────┐ ┌──────────┐ │ Callback │ ──▶ │ Promise │ ──▶ │ async/ │ │ （回调） │ │ （承诺） │ │ await │ └──────────┘ └──────────┘ └──────────┘ 回调地狱 链式调用 同步风格

数组高阶方法

掌握数组的高阶函数可以写出更简洁、更具表达力的代码。

JavaScript 
const numbers = [1, 2, 3, 4, 5];

// map - 映射为新数组
const doubled = numbers.map(n => n * 2);  // [2, 4, 6, 8, 10]

// filter - 过滤元素
const evens = numbers.filter(n => n % 2 === 0);  // [2, 4]

// reduce - 归约为单个值
const sum = numbers.reduce((acc, n) => acc + n, 0);  // 15

// find - 查找第一个匹配元素
const firstEven = numbers.find(n => n % 2 === 0);  // 2

// some - 是否有满足条件的元素
const hasLarge = numbers.some(n => n > 4);  // true

// every - 是否全部满足条件
const allPositive = numbers.every(n => n > 0);  // true

// 链式调用
const result = numbers
  .filter(n => n > 2)
  .map(n => n * 10)
  .reduce((a, b) => a + b, 0);  // (3*10) + (4*10) + (5*10) = 120
