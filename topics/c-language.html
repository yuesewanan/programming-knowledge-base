C 语言

编程语言的基石。C 语言提供了对内存的精细控制，是操作系统、嵌入式系统和性能敏感应用的首选语言。

基础语法

C
// 基本结构
#include <stdio.h>

int main() {
    // 变量声明
    int age = 25;
    float pi = 3.14f;
    double price = 19.99;
    char grade = 'A';
    char name[] = "Alice";
    const int MAX = 100;

    // 输入输出
    printf("年龄: %d, 分数: %.2f\n", age, price);
    scanf("%d", &age);     // 注意 & 取地址

    // 控制流
    if (age >= 18) {
        printf("成年\n");
    } else {
        printf("未成年\n");
    }

    // 循环
    for (int i = 0; i < 5; i++) {
        printf("%d\n", i);
    }

    return 0;
}

// 函数定义
int add(int a, int b) {
    return a + b;
}
📘 注意C 语言需要手动管理内存，没有垃圾回收机制。printf 和 scanf 使用格式化字符串控制输入输出。

指针

指针是 C 语言的精髓，直接操作内存地址。

C
// 指针基础
int x = 42;
int *p = &x;        // p 存储 x 的地址

printf("x 的值: %d\n", x);           // 42
printf("x 的地址: %p\n", &x);
printf("p 的值: %p\n", p);           // 地址
printf("p 指向的值: %d\n", *p);      // 42, 解引用

*p = 100;              // 通过指针修改变量
printf("%d\n", x);       // 100

// 指针与函数参数 (传引用)
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int a = 1, b = 2;
swap(&a, &b);    // a=2, b=1

// NULL 指针
int *ptr = NULL;
if (ptr != NULL) {
    // 安全使用
}

// 指针运算
int arr[] = {10, 20, 30};
int *p_arr = arr;
printf("%d\n", *(p_arr + 1));  // 20, 指针偏移
⚠️ 指针常见错误使用未初始化的指针（野指针）、访问已释放的内存（悬空指针）、缓冲区溢出——这些是 C 语言最常见的 Bug 来源。

内存管理

使用标准库函数 malloc/free 进行动态内存分配。

C
#include <stdlib.h>
#include <stdio.h>

int main() {
    // 分配内存
    int *arr = (int *)malloc(5 * sizeof(int));

    if (arr == NULL) {
        printf("内存分配失败\n");
        return 1;
    }

    // 使用内存
    for (int i = 0; i < 5; i++) {
        arr[i] = i * 2;
    }

    // calloc - 分配并初始化为0
    int *zeroed = (int *)calloc(5, sizeof(int));

    // realloc - 重新分配大小
    arr = (int *)realloc(arr, 10 * sizeof(int));

    // 释放内存!
    free(arr);
    free(zeroed);

    // 释放后置 NULL 防止悬空指针
    arr = NULL;

    return 0;
}
⚠️ 内存泄漏每次 malloc 必须有对应的 free。使用 Valgrind 等工具检测内存泄漏。

结构体与联合体

C
// 结构体定义
struct Student {
    char name[50];
    int age;
    float score;
};

// typedef 简化类型名
typedef struct {
    char title[100];
    char author[50];
    int year;
} Book;

int main() {
    // 声明和初始化
    struct Student s1 = {"Alice", 20, 85.5};

    // 访问成员
    printf("名字: %s, 年龄: %d\n", s1.name, s1.age);

    // 结构体指针
    struct Student *p = &s1;
    printf("分数: %.1f\n", p->score);  // 使用 -> 访问

    // 联合体 (所有成员共享同一内存)
    union Data {
        int i;
        float f;
        char str[20];
    };

    return 0;
}

文件操作

C
#include <stdio.h>

int main() {
    // 写入文件
    FILE *fp = fopen("data.txt", "w");
    if (fp == NULL) {
        printf("文件打开失败\n");
        return 1;
    }

    fprintf(fp, "Hello, World!\n");
    fprintf(fp, "数字: %d\n", 42);
    fclose(fp);     // 关闭文件

    // 读取文件
    fp = fopen("data.txt", "r");
    char buffer[256];
    while (fgets(buffer, sizeof(buffer), fp)) {
        printf("%s", buffer);
    }
    fclose(fp);

    // 二进制文件读写
    typedef struct { int id; char name[32]; } Record;
    Record r = {1, "Alice"};

    fp = fopen("records.bin", "wb");
    fwrite(&r, sizeof(Record), 1, fp);
    fclose(fp);

    return 0;
}
